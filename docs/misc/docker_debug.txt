import requests
import os
host = os.getenv('STORAGE_EMULATOR_HOST', 'http://gcs:4443')
# The internal path for fake-gcs-server to get a file:
test_url = f'{host}/storage/v1/b/data/o/movies-00000-of-00001.array_record?alt=media'
print(f'Attempting to reach: {test_url}')
try:
    response = requests.get(test_url, timeout=5)
    if response.status_code == 200:
        print(f'SUCCESS: Reached file. First 20 bytes: {response.content[:20].hex()}')
    else:
        print(f'FAILURE: Server responded with {response.status_code}')
        print(f'Response body: {response.text}')
except Exception as e:
    print(f'NETWORK ERROR: Could not connect to {host}. Error: {e}')
"
